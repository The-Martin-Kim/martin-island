#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    vector<int> A(7, 0);
    
    for(int i = 0; i < 7; i++){
        cin >> A[i];
    }
    
    for(int i = 0; i < 7; i++){
        for(int j = i + 1; j < 7; j++){
            for(int k = j + 1; k < 7; k++){
                bool r1 = find(A.begin(), A.end(), A[i] + A[j]) != A.end();
                bool r2 = find(A.begin(), A.end(), A[j] + A[k]) != A.end();
                bool r3 = find(A.begin(), A.end(), A[k] + A[i]) != A.end();
                bool r4 = find(A.begin(), A.end(), A[i] + A[j] + A[k]) != A.end();
                
                if (r1 && r2 && r3 && r4){
                    vector<int> result = {A[i], A[j], A[k]};
                    sort(result.begin(), result.end());
                    
                    for(int i = 0; i < 3; i++){
                        cout << result[i] << " ";
                    }
                    cout << "\n";
                    
                    return 0;
                }
            }
        }
    }
    
    return 0;
}
