#include <iostream>
#include <string>
#include <vector>
using namespace std;

vector<int> ACGT(4, 0);
vector<int> newACGT(4, 0);

void add(char c){
    switch (c){
        case 'A':
            newACGT[0]++;
            break;
        case 'C':
            newACGT[1]++;
            break;
        case 'G':
            newACGT[2]++;
            break;
        case 'T':
            newACGT[3]++;
            break;
    }
}

void remove(char c){
    switch (c){
        case 'A':
            newACGT[0]--;
            break;
        case 'C':
            newACGT[1]--;
            break;
        case 'G':
            newACGT[2]--;
            break;
        case 'T':
            newACGT[3]--;
            break;
    }
}

bool flag(){
    for (int i = 0; i < 4; i++){
        if (ACGT[i] <= newACGT[i]){
            continue;
        }
        else{
            return false;
        }
    }
    
    return true;
}

int main()
{
    int S, P;
    cin >> S >> P;
    
    string DNA;
    cin >> DNA;
    
    for (int i = 0; i < 4; i++){
        cin >> ACGT[i];
    }
    
    for (int i = 0; i < P; i++){
        add(DNA[i]);
    }
    
    int result = 0;
    if (flag()){
        result++;
    }
    
    for (int i = 1; i < S - P + 1; i++){
        remove(DNA[i - 1]);
        add(DNA[i + P - 1]);
        
        if (flag()){
            result++;
        }
    }
    
    cout << result << endl;
   
    return 0;
}
