    for (int i = 2; i <= sqrt(N); i++){
        if (prime[i] == 0){
            continue;
        }
        
        for (int j = i + i; j <= N; j = j + i){
            prime[j] = 0;
        }
    }
    
    for (int i = M; i <= N; i++){
        if (prime[i] != 0){
            cout << prime[i] << '\n';
        }
    }
    
    return 0;
}
