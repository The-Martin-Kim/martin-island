    vector<int> R(M, 0);
    for (int i = 0; i < M; i++){
        int start, end;
        cin >> start >> end;
        R[i] = S[end] - S[start - 1];
    }
    
    for (int i = 0; i < M; i++){
        cout << R[i] << '\n';
    }
    
    return 0;
}
