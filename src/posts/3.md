```c++
#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int N, M;
    cin >> N >> M;
    
    vector<int> A(N + 1, 0);
    for (int i = 1; i <= N; i++){
        cin >> A[i];
    }
    
    vector<int> S(M, 0);
    for (int i = 0; i < M; i++){
        int start, end;
        cin >> start >> end;
        for (int j = start; j <= end; j++){
            S[i] += A[j];
        }
    }
    
    for (int i = 0; i < M; i++){
        cout << S[i] << '\n';
    }
    
    return 0;
}

#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int N, M;
    cin >> N >> M;
    
    vector<int> S(N + 1, 0);
    for (int i = 1; i <= N; i++){
        int temp;
        cin >> temp;
        S[i] = S[i - 1] + temp;
    }
    
    vector<int> R(M, 0);
    for (int i = 0; i < M; i++){
        int start, end;
        cin >> start >> end;
        R[i] = S[end] - S[start - 1];
    }
    
    for (int i = 0; i < M; i++){
        cout << R[i] << '\n';
    }
    
    return 0;
}
```
