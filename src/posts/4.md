<h4><strong>📓 비효율적인 풀이</strong></h4>

<div class="code-block " style="margin-bottom: 20px">

```c++
#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int N;
    cin >> N;
    
    int count = 0;
    for (int i = 1; i <= N; i++){
        int temp = 0;
        for (int j = i; j <= N; j++){
            temp += j;
            if (temp == N){
                count++;
                break;
            }
            
            if (temp > N){
                break;
            }
        }
    }
    
    cout << count << '\n';
    
    return 0;
}
```

<h4><strong>💡 효율적인 풀이</strong></h4>

<div class="code-block">

```c++
#include <iostream>
using namespace std;

int main()
{
    int N;
    cin >> N;
    
    int start = 1;
    int end = 1;
    int count = 1;
    int total = 1;
    
    while (end != N){
        if (total > N){
            total -= start;
            start++;
        }
        else if (total < N){
            end++;
            total += end;
        }
        else if (total == N){
            count++;
            end++;
            total += end;
        }
    }
    
    cout << count << '\n';
    
    return 0;
}
```
</div>
