#include <iostream>

using namespace std;

    ios::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> N;
    cin >> M;

    sort(A.begin(), A.end());
    
    int count = 0;
    int i = 0;
    int j = N - 1;
    
    while (i < j){
        if (A[i] + A[j] > M){
            j--;
        }
        
        else if (A[i] + A[j] < M){
            i++;
        }
        
        else if (A[i] + A[j] == M){
            count++;
            i++;
            j--;
        }
    }
    
    cout << count << '\n';
    
    return 0;
}
