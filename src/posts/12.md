#include <iostream>
#include <vector>
using namespace std;

int main()
{   
    int N;
    cin >> N;
    
    vector<int> distance(N - 1, 0);
    vector<int> gasPrice(N, 0);
    
    for (int i = 0; i < N - 1; i++){
        cin >> distance[i];
    }
    
    for (int i = 0; i < N; i++){
        cin >> gasPrice[i];
    }
    
    int minPrice = gasPrice[0];
    long long totalPrice = 0;

    for (int i = 0; i < N - 1; i++){
        if (minPrice > gasPrice[i]){
            minPrice = gasPrice[i];
        }
        
        totalPrice += static_cast<long long>(minPrice) * distance[i];
    }
    
    cout << totalPrice << '\n';

    return 0;
}
