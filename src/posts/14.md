#include <iostream>
#include <string>
using namespace std;

int ACGT[4];
int new_ACGT[4] = {0, 0, 0, 0};

void add(char c){
    if (c == 'A'){
        new_ACGT[0]++;
    }
    else if (c == 'C'){
        new_ACGT[1]++;
    }
    else if (c == 'G'){
        new_ACGT[2]++;
    }
    else if (c == 'T'){
        new_ACGT[3]++;
    }
}

void remove(char c){
    if (c == 'A'){
        new_ACGT[0]--;
    }
    else if (c == 'C'){
        new_ACGT[1]--;
    }
    else if (c == 'G'){
        new_ACGT[2]--;
    }
    else if (c == 'T'){
        new_ACGT[3]--;
    }
}

bool pw_check(){
    for (int i = 0; i < 4; i++){
        if (new_ACGT[i] >= ACGT[i]){
            continue;
        }
        else{
            return false;
        }
    }
    
    return true;
}

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    
    int S, P;

    for (int i = 0; i < 4; i++){

        add(DNA[i - 1 + P]);

            count++;
        }
    }

}
