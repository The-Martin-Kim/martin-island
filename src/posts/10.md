#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int N;
vector<int> A(100000, 0);

int binary_search(int target){
    int start = 0;
    int mid;
    int end = N - 1;
    
    while (start <= end){
        mid = (start + end) / 2;
        
        if (A[mid] == target){
            return 1;
        }
        else if (A[mid] > target){
            end = mid - 1;
        }
        else if (A[mid] < target){
            start = mid + 1;
        }
    }
    
    return 0;
}

int main()
{
    cin >> N;
    
    for (int i = 0; i < N; i++){
        cin >> A[i];
    }
    
    sort(A.begin(), A.begin() + N);
    
    int M;
    cin >> M;
    
    vector<int> result(M, 0);
    for (int i = 0 ; i < M; i++){
        int target;
        cin >> target;
        
        result[i] = binary_search(target);
    }
    
    for (int i = 0; i < M; i++){
        cout << result[i] << '\n';
    }

    return 0;
}
