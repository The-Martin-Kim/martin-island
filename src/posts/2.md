<h4><strong>📓 일반적인 풀이</strong></h4>

<div class="code-block " style="margin-bottom: 20px">

```c++
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int N;
    cin >> N;
    
    vector<float> scores(N, 0);
    for(int i = 0; i < N; i++){
        cin >> scores[i];
    }
    
    int max = *max_element(scores.begin(), scores.end());
    
    float sum = 0;
    for(int i = 0; i < N; i++){
        scores[i] = scores[i] / max * 100;
        sum += scores[i];
    }
    
    float avg = sum / N;
    
    cout << avg << '\n';
    
    return 0;
}
```

<h4><strong>💡 똑똑한 풀이</strong></h4>

<div class="code-block">

```c++
#include <iostream>
using namespace std;

int main()
{
    int N;
    cin >> N;
    
    float sum = 0.0;
    int max = -1;
    for(int i = 0; i < N; i++){
        int temp;
        cin >> temp;
        
        if (temp > max){
            max = temp;
        }
        
        sum += temp;
    }
    
    float avg = sum / max * 100 / N;
    
    cout << avg << '\n';
    
    return 0;
}
```
