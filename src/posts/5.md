    int i = 0;
    int j = N - 1;

        if (A[i] + A[j] > M){
            j--;
        }
        
        else if (A[i] + A[j] < M){
            i++;
        }
        
        else if (A[i] + A[j] == M){
            count++;
            i++;
            j--;
        }
    }
    
    cout << count << '\n';
    
    return 0;
}
